
if(${CMAKE_SYSTEM_NAME} STREQUAL XCORE_XS3A)
    ## Create Explorer Board target
    add_library(xcore_sdk_rtos_board_support_config_xcore_ai_explorer INTERFACE)
    target_sources(xcore_sdk_rtos_board_support_config_xcore_ai_explorer
        INTERFACE
            XCORE-AI-EXPLORER/platform/aic3204.c
            XCORE-AI-EXPLORER/platform/app_pll_ctrl.c
            XCORE-AI-EXPLORER/platform/driver_instances.c
            XCORE-AI-EXPLORER/platform/platform_init.c
            XCORE-AI-EXPLORER/platform/platform_start.c
    )
    target_include_directories(xcore_sdk_rtos_board_support_config_xcore_ai_explorer
        INTERFACE
            XCORE-AI-EXPLORER
    )
    target_link_libraries(xcore_sdk_rtos_board_support_config_xcore_ai_explorer
        INTERFACE
            sdk::core
            sdk::rtos_freertos
            sdk::rtos::audio_drivers
    )
    target_compile_definitions(xcore_sdk_rtos_board_support_config_xcore_ai_explorer
        INTERFACE
            XCOREAI_EXPLORER=1
            PLATFORM_SUPPORTS_TILE_0=1
            PLATFORM_SUPPORTS_TILE_1=1
            PLATFORM_SUPPORTS_TILE_2=0
            PLATFORM_SUPPORTS_TILE_3=0
            USB_TILE_NO=0
            USB_TILE=tile[USB_TILE_NO]

            MIC_ARRAY_CONFIG_MCLK_FREQ=24576000
            MIC_ARRAY_CONFIG_PDM_FREQ=3072000
            MIC_ARRAY_CONFIG_SAMPLES_PER_FRAME=256
            MIC_ARRAY_CONFIG_MIC_COUNT=2
            MIC_ARRAY_CONFIG_CLOCK_BLOCK_A=XS1_CLKBLK_1
            MIC_ARRAY_CONFIG_CLOCK_BLOCK_B=XS1_CLKBLK_2
            MIC_ARRAY_CONFIG_PORT_MCLK=PORT_MCLK_IN
            MIC_ARRAY_CONFIG_PORT_PDM_CLK=PORT_PDM_CLK
            MIC_ARRAY_CONFIG_PORT_PDM_DATA=PORT_PDM_DATA
            # DEBUG_PRINT_ENABLE_RTOS_MIC_ARRAY=1
    )

    ## Create an alias
    add_library(sdk::rtos_bsp::xcore_ai_explorer ALIAS xcore_sdk_rtos_board_support_config_xcore_ai_explorer)
endif()
